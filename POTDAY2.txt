// Problem:- remove the vowels from the string

class Solution{
public:	
		
	string removeVowels(string S) 
	{
	    int i;
	  
	    for(i=0;i<S.length();i++)
	    {
	       
	    if(S[i]=='a' || S[i]=='e' || S[i]=='i' || S[i]=='o' || S[i]=='u')
	    {
	    cout<<"";
	        
	    }
	    else
	    {
	        cout<<S.at(i);
	        
	    
	        
	    }
	        
	    }
	    
	}
};
