#include<iostream>
#include <bits/stdc++.h>
using namespace std;
int main()
{
    int i,j,k,n;
    cout<<"Enter the number : "<<endl;
    cin>>n;
    for(i=1;i<=n;i++)
    {
        for(j=n;j>=1;j--)
        {
            if(i>=j)
            cout<<"* ";
            else
            cout<<" ";
        }
        cout<<endl;

    }
    return 0;
    
}
